#include <stdio.h>
#include <stdlib.h>

typedef struct node
{
	char x;
	struct node *left;
	struct node *right;
}NODE;

NODE *newNode(char character)
{
	NODE *node = (NODE*) malloc (sizeof(NODE));
	node->x = character;
	node->left = NULL;
	node->right = NULL;
	return node;
}

NODE *insertNode(NODE *node, char character)
{
	if(node == NULL)
	{
		return newNode(character);
	}
	else if(character < node->x)
	{
		node->left = insertNode(node->left,character);
	}
	else if(character > node->x)
	{
		node->right = insertNode(node->right,character);
	}
	return node;
}


void Preorder(NODE *node)
{
	if(node != NULL)
	{
		printf("%c ",node->x);
		Preorder(node->left);
		Preorder(node->right);
	}
}

void Inorder(NODE *node)
{
	if(node != NULL)
	{
		Inorder(node->left);
		printf("%c ",node->x);
		Inorder(node->right);
	}
}

void Postorder(NODE *node)
{
	if(node != NULL)
	{
		Postorder(node->left);
		Postorder(node->right);
		printf("%c ",node->x);
	}
}

int main()
{
	NODE *node = NULL;
	int rootCheck = 0;
	char character;
	printf("Unesite slova u stablo: (Ne zaboravite tocku na kraju rijeci npr. dominikvidovic. )\n");
	do
	{
		scanf("%c",&character);
		if(character == 46)  //46 ASCII vrijednost za fullstop ili tocku
		{
			break;  //ako se unese tocka na kraju
		}
		if(rootCheck == 0)
		{
			node = insertNode(node,character);
		}			
		else
		{
			insertNode(node,character);			
		}
		rootCheck = 1;
	}while(character != 46);
	
	printf("PREORDER:\n");
	Preorder(node);
	printf("\nINORDER:\n");
	Inorder(node);
	printf("\nPOSTORDER:\n");
	Postorder(node);

	return 0;
}
